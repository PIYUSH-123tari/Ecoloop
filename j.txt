  const userId = localStorage.getItem("userId");

  if (!userId) {
    alert("User not logged in");
    return;
  }

  try {
    const response = await fetch(`http://localhost:5000/users/${userId}`);

    if (!response.ok) {
      throw new Error("Failed to fetch user data");
    }

    const user = await response.json();

    // Fill form fields
    document.querySelector("input[type='text']").value = user.name || "";
    document.querySelector("input[type='email']").value = user.email || "";
    document.querySelector("input[type='tel']").value = user.phone || "";
    document.querySelector("select").value = user.region_Id || "";

    // Show photo if exists
    if (user.photo) {
      document.querySelector(".avatar").src =
        `http://localhost:5000/uploads/${user.photo}`;
    }

  } catch (error) {
    console.error("Error fetching profile:", error);
  }

});
